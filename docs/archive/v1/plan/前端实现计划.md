### 2. API 层 (api/)

#### client.ts
```typescript
import axios from 'axios'

const apiClient = axios.create({
  baseURL: 'http://localhost:8080/api',
  timeout: 60000,  // LLM 调用可能较慢
  headers: {
    'Content-Type': 'application/json'
  }
})

// 响应拦截器：统一处理 Result 格式
apiClient.interceptors.response.use(
  response => response.data.data,
  error => {
    console.error('API Error:', error)
    return Promise.reject(error)
  }
)

export default apiClient
```

#### index.ts
```typescript
import apiClient from './client'
import type { Knowledge, TrainingRecord, EvaluationResult } from '../types'

export const knowledgeAPI = {
  getList: () => apiClient.get<Knowledge[]>('/knowledge'),
  getById: (id: number) => apiClient.get<Knowledge>(`/knowledge/${id}`),
  create: (data: { title: string, content: string }) =>
    apiClient.post<Knowledge>('/knowledge', data),
  update: (id: number, data: { title: string, content: string }) =>
    apiClient.put<Knowledge>(`/knowledge/${id}`, data),
  delete: (id: number) => apiClient.delete(`/knowledge/${id}`)
}

export const trainingAPI = {
  start: (knowledgeId: number) =>
    apiClient.post<{ question: string }>('/training/start', { knowledgeId }),
  submit: (data: {
    knowledgeId: number
    question: string
    answer: string
  }) => apiClient.post<EvaluationResult>('/training/submit', data),
  getHistory: (knowledgeId: number) =>
    apiClient.get<TrainingRecord[]>(`/training/history/${knowledgeId}`),
  getAllHistory: () =>
    apiClient.get<TrainingRecord[]>('/training/history')
}
```

### 5. 公共组件

#### MasteryBadge.vue

```vue
<template>
  <div class="mastery-badge" :class="levelClass">
    {{ level }}
  </div>
</template>

<script setup lang="ts">
const props = defineProps<{
  mastery: number
}>()

const level = computed(() => {
  if (props.mastery >= 80) return '精通'
  if (props.mastery >= 60) return '熟练'
  if (props.mastery >= 40) return '掌握'
  return '入门'
})

const levelClass = computed(() => {
  if (props.mastery >= 80) return 'expert'
  if (props.mastery >= 60) return 'proficient'
  if (props.mastery >= 40) return 'competent'
  return 'beginner'
})
</script>
```

## ⚠️ 前端约束

1. **禁止过度组件化** - V1 不追求完美的组件抽象
2. **禁止复杂动画** - 保持简洁
3. **禁止使用 UI 框架** - 手写简单样式
4. **必须加载状态** - LLM 调用需要明确的加载提示
5. **必须错误处理** - 所有 API 调用都要处理错误
